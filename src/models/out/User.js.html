import { model, Schema } from "mongoose";

/**
 * @fileoverview Mongoose schema and model definition for the User collection.
 */

/**
 * User Schema
 * @typedef {Object} UserSchema
 * @property {String} userKey - The unique key for the user.
 * @property {String} cnic - The unique CNIC (Computerized National Identity Card) number for the user.
 * @property {String} email - The unique email address of the user.
 * @property {String} [rank] - The rank of the user (optional).
 * @property {String} [badgeNumber] - The badge number of the user (optional).
 * @property {ObjectId} PS - The reference to the police station associated with the user.
 */

const UserSchema = new Schema({
  userKey: {
    type: String,
    required: true,
  },
  cnic: {
    type: String,
    required: true,
    unique: true,
  },
  email: {
    type: String,
    unique: true,
    required: true,
  },
  rank: String,
  badgeNumber: String,
  PS: {
    type: Schema.Types.ObjectId,
    name: Schema.name,
    ref: "policestations",
  },
});

/**
 * User model
 * @typedef {Object} UserModel
 * @property {Function} find - Mongoose method to find documents.
 * @property {Function} findById - Mongoose method to find a document by ID.
 * @property {Function} create - Mongoose method to create a new document.
 * @property {Function} update - Mongoose method to update a document.
 * @property {Function} delete - Mongoose method to delete a document.
 */

/**
 * @type {UserModel}
 */
const User = model("user", UserSchema);

export default User;
